# Fix Chat IA — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace fragmented `/api/ai/search` and `/api/ai/chart` calls with a unified `/api/ai/ask` endpoint that classifies intent, supports conversational followups via `lastFilter`, suggests alternatives on 0-result queries, and presents a readable header — eliminating the "chat siempre responde lo mismo" bug and the WCAG-failing subtitle.

**Architecture:** Server-side pipeline — heuristic classifier → LLM classifier fallback (only `unknown` cases) → search/chart parser (with optional `lastFilter` injection) → Postgres query → neighbor analysis on 0-results → uniform response with discriminated `kind`. Frontend becomes a thin renderer over `kind`. Existing `/search` and `/chart` endpoints become deprecated wrappers that delegate to the new orchestrator.

**Tech Stack:** NestJS 10, Prisma 5 + Postgres 16, zod, Jest (api), Next.js 15 + TanStack Query, Tailwind, Vitest + RTL (web), Playwright (e2e).

**Spec:** `docs/superpowers/specs/2026-05-03-fix-chat-ia-design.md`

```typescript
  latencyMs: z.number().int().nonnegative(),

  classifier: z.enum(['heuristic', 'llm']),

  count: z.number().int().nonnegative(),

  filter: AiFilterSchema,

});
```

```typescript
import { Injectable } from '@nestjs/common';

export type Intent = 'greeting' | 'search' | 'chart' | 'unknown';

const GREETING_RE = /^(hola|holi|holis|buenas|hey|hi|hello|gracias|chao|ok|listo|dale)\b/i;
const CHART_KW    = /\b(grafico|gráfico|tendencia|compara|distribuci|evoluci|histograma|pie|barras|por (mes|año|psr|abc))\b/i;
const DOMAIN_KW   = /\b(vencid|psr|hh|preventiv|manto|equipo|pm0\d|2026|2027|abc|ejecutad|planificad|denomi|posici|ticket|antic)\b/i;

@Injectable()
export class HeuristicClassifier {
  classify(prompt: string): Intent {
    const p = (prompt ?? '').trim().toLowerCase();
    if (!p) return 'greeting';
    if (GREETING_RE.test(p) && p.length < 30) return 'greeting';
    if (CHART_KW.test(p)) return 'chart';
    if (DOMAIN_KW.test(p)) return 'search';
    return 'unknown';
  }
}
```

```typescript
import { Injectable } from '@nestjs/common';
import type { LlmClassifierClient } from '../llm.client';

export interface LlmClassification { kind: 'search'|'chart'|'clarify'; confidence: number; }

const VALID_KINDS = new Set(['search', 'chart', 'clarify']);
const MIN_CONFIDENCE = 0.6;

@Injectable()
export class LlmClassifier {
  constructor(private client: LlmClassifierClient) {}

  async classify(prompt: string, sessionContext?: { lastFilter?: unknown; lastMode?: 'search'|'chart' }): Promise<LlmClassification> {
    let raw: string;
    try {
      const res = await this.client.call(prompt, sessionContext);
      raw = res.raw;
    } catch {
      return { kind: 'clarify', confidence: 0 };
    }
    let parsed: any;
    try { parsed = JSON.parse(raw); } catch { return { kind: 'clarify', confidence: 0 }; }
    if (!parsed || typeof parsed !== 'object') return { kind: 'clarify', confidence: 0 };
    if (!VALID_KINDS.has(parsed.kind)) return { kind: 'clarify', confidence: 0 };
    const conf = typeof parsed.confidence === 'number' ? parsed.confidence : 0;
    if (conf < MIN_CONFIDENCE) return { kind: 'clarify', confidence: conf };
    return { kind: parsed.kind, confidence: conf };
  }
}
```

```typescript
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../../prisma/prisma.service';
import type { Suggestion } from '../dto/ask.dto';

const STATIC_FALLBACK: Suggestion[] = [
  { type: 'prompt', label: 'Vencidas mes', prompt: 'vencidas mes' },
  { type: 'prompt', label: 'HH proyectadas 2026', prompt: 'gráfico HH 2026' },
  { type: 'prompt', label: 'Plan manto. anuales', prompt: 'plan manto anuales' },
];

@Injectable()
export class GreetingService {
  constructor(private prisma: PrismaService) {}

  async build(userId: string): Promise<{ message: string; suggestions: Suggestion[] }> {
    const suggestions: Suggestion[] = [];
    try {
      const topPsr = await this.prisma.execution.groupBy({
        by: ['psr'],
        where: { dueDate: { gte: new Date(), lte: new Date(Date.now() + 30 * 86400_000) } },
        _count: { _all: true },
        orderBy: { _count: { psr: 'desc' } },
        take: 1,
      } as any);
      if (topPsr[0]?.psr) {
        suggestions.push({ type: 'prompt', label: `PSR ${topPsr[0].psr} próx. 4 sem.`, prompt: `PSR ${topPsr[0].psr} próximas 4 semanas` });
      }
    } catch { /* graceful fallback */ }

    suggestions.push({ type: 'prompt', label: 'Vencidas mes', prompt: 'vencidas mes' });
    suggestions.push({ type: 'prompt', label: 'HH 2026', prompt: 'gráfico HH 2026' });

    const finalSugs = suggestions.length >= 3 ? suggestions.slice(0, 3) : STATIC_FALLBACK;
    return { message: 'Hola. ¿Qué revisamos?', suggestions: finalSugs };
  }
}
```

```typescript
  countWithFilter(filter: Partial<AiFilter>): Promise<number>;
}

const DROPPABLE_KEYS: Array<keyof AiFilter> = [
  'psr',
  'abc',
  'frecuenciaCodigo',
  'centroPlanificacion',
  'equipo',
  'ubicacionTecnica',
  'yearFrom',
  'yearTo',
  'monthFrom',
  'monthTo',
  'onlyOverdue',
  'onlyUpcomingDays',
  'q',
];

@Injectable()
export class NeighborAnalyzer {
  constructor(private counter: NeighborCounter) {}

  async analyze(filter: AiFilter): Promise<Suggestion[]> {
    const candidates: Array<{ label: string; filterDelta: Partial<AiFilter>; count: number }> = [];

    for (const key of DROPPABLE_KEYS) {
      if (filter[key] === undefined) continue;
      const next = { ...filter };
      delete (next as any)[key];
      // Pair year drops
      if (key === 'yearFrom' || key === 'yearTo') {
        delete (next as any).yearFrom;
        delete (next as any).yearTo;
      }
      if (key === 'monthFrom' || key === 'monthTo') {
        delete (next as any).monthFrom;
        delete (next as any).monthTo;
      }
      const count = await this.counter.countWithFilter(next);
      if (count > 0) {
        candidates.push({
          label: this.labelFor(key, filter),
          filterDelta: this.deltaFor(key),
          count,
        });
      }
    }

    candidates.sort((a, b) => b.count - a.count);
    return candidates.slice(0, 3).map((c) => ({ type: 'filterDelta' as const, ...c }));
  }

  private labelFor(key: keyof AiFilter, filter: AiFilter): string {
    const value = filter[key];
    switch (key) {
      case 'psr':                 return `Quitar PSR ${value}`;
      case 'abc':                 return `Quitar ABC=${value}`;
      case 'yearFrom': case 'yearTo': return 'Cualquier año';
      case 'monthFrom': case 'monthTo': return 'Cualquier mes';
      case 'frecuenciaCodigo':    return `Quitar frecuencia ${value}`;
      case 'centroPlanificacion': return `Quitar centro ${value}`;
      case 'equipo':              return `Quitar equipo ${value}`;
      case 'ubicacionTecnica':    return `Quitar ubicación`;
      case 'onlyOverdue':         return `Incluir no vencidas`;
      case 'onlyUpcomingDays':    return `Cualquier rango futuro`;
      case 'q':                   return `Quitar texto libre`;
      default:                    return `Quitar ${String(key)}`;
    }
  }

  private deltaFor(key: keyof AiFilter): Partial<AiFilter> {
    const d: Partial<AiFilter> = {};
    if (key === 'yearFrom' || key === 'yearTo') {
      (d as any).yearFrom = undefined;
      (d as any).yearTo = undefined;
    } else if (key === 'monthFrom' || key === 'monthTo') {
      (d as any).monthFrom = undefined;
      (d as any).monthTo = undefined;
    } else {
      (d as any)[key] = undefined;
    }
    return d;
  }
}
```

```typescript
  userId: string;
  sessionId: string;
  prompt: string;
  kind: string;
  filter?: unknown;

  model?: string;

export class AiQueryLogService {

  constructor(private prisma: PrismaService) {}

    try {

          sessionId: args.sessionId,
          promptHash: createHash('sha256').update(args.prompt).digest('hex'),
          kind: args.kind,
          filter: args.filter as any,
          count: args.count ?? null,
          latencyMs: args.latencyMs,
          model: args.model ?? null,
        },
      });
    } catch (e) {
      this.log.warn(`ai_query_log_write_failed: ${(e as Error).message}`);
    }
  }
}
```

```typescript
import { Injectable } from '@nestjs/common';
import { Role } from '@prisma/client';
import { HeuristicClassifier, type Intent } from './classifier/heuristic.classifier';
import { LlmClassifier } from './classifier/llm.classifier';
import { GreetingService } from './templates/greeting.service';
import { AiSearchService } from './ai-search.service';
import { ChartBuilderService } from './chart.service';
import { NeighborAnalyzer } from './neighbors/neighbor-analyzer.service';
import { AiQueryLogService } from './audit/ai-query-log.service';
import type { AskRequest, AskResponse, Suggestion } from './dto/ask.dto';

interface AskCtx { ip: string; userAgent: string; sessionId: string; requestId: string; }

@Injectable()
export class AiAskService {
  constructor(
    private heuristic: HeuristicClassifier,
    private llmClassifier: LlmClassifier,
    private greeting: GreetingService,
    private search: AiSearchService,
    private chart: ChartBuilderService,
    private neighbors: NeighborAnalyzer,
    private logSvc: AiQueryLogService,
  ) {}

  async ask(user: { id: string; role: Role }, body: AskRequest, ctx: AskCtx): Promise<AskResponse> {
    const start = Date.now();
    const baseMeta = { requestId: ctx.requestId, classifier: 'heuristic' as 'heuristic' | 'llm' };

    let intent: Intent = this.heuristic.classify(body.prompt);

    // Greeting can never be overridden
    if (intent === 'greeting') {
      const g = await this.greeting.build(user.id);
      const meta = { ...baseMeta, model: 'template', latencyMs: Date.now() - start, parser: 'heuristic' as const };
      const resp: AskResponse = { kind: 'greeting', payload: g, meta };
      await this.logSvc.write({ userId: user.id, sessionId: ctx.sessionId, prompt: body.prompt, kind: 'greeting', latencyMs: meta.latencyMs });
      return resp;
    }

    // Override pisa cuando intent es search/chart/unknown
    if (body.override) {
      const forced = await this.runQuery(user, body, ctx, body.override);
      const resp: AskResponse = { ...forced, mode: 'forced' };
      await this.logSvc.write({ userId: user.id, sessionId: ctx.sessionId, prompt: body.prompt, kind: resp.kind, filter: (resp.payload as any).filter, count: (resp.payload as any).count, latencyMs: resp.meta.latencyMs, model: resp.meta.model });
      return resp;
    }

    // Unknown → LLM classifier
    if (intent === 'unknown') {
      const cls = await this.llmClassifier.classify(body.prompt, body.sessionContext);
      baseMeta.classifier = 'llm';
      if (cls.kind === 'clarify') {
        const meta = { ...baseMeta, model: 'classifier', latencyMs: Date.now() - start, parser: 'heuristic' as const };
        const resp: AskResponse = {
          kind: 'clarify',
          payload: { message: 'No entendí. Reformula con keywords como "vencidas", "PSR", "HH 2026".', suggestions: [
            { type: 'prompt', label: 'Vencidas mes', prompt: 'vencidas mes' },
            { type: 'prompt', label: 'HH 2026', prompt: 'gráfico HH 2026' },
          ] },
          meta,
        };
        await this.logSvc.write({ userId: user.id, sessionId: ctx.sessionId, prompt: body.prompt, kind: 'clarify', latencyMs: meta.latencyMs });
        return resp;
      }
      intent = cls.kind;
    }

    // search or chart
    const result = await this.runQuery(user, body, ctx, intent as 'search' | 'chart');
    const resp: AskResponse = { ...result, mode: 'detected' };
    await this.logSvc.write({ userId: user.id, sessionId: ctx.sessionId, prompt: body.prompt, kind: resp.kind, filter: (resp.payload as any).filter, count: (resp.payload as any).count, latencyMs: resp.meta.latencyMs, model: resp.meta.model });
    return resp;
  }

  private async runQuery(
    user: { id: string; role: Role },
    body: AskRequest,
    ctx: AskCtx,
    forced: 'search' | 'chart',
  ): Promise<Extract<AskResponse, { kind: 'search' } | { kind: 'chart' }>> {
    if (forced === 'search') {
      const r = await this.search.search(user, body.prompt, { ip: ctx.ip, userAgent: ctx.userAgent }, body.sessionContext?.lastFilter);
      const meta = { model: r._meta.model, latencyMs: r._meta.latencyMs, parser: r._meta.parser, classifier: 'heuristic' as const, requestId: ctx.requestId };
      const suggestions: Suggestion[] = r.count === 0 ? await this.neighbors.analyze(r.filter) : [];
      return { kind: 'search', payload: { count: r.count, rows: r.rows ?? [], filter: r.filter, suggestions: suggestions.length ? suggestions : undefined }, mode: 'detected' as const, meta };
    } else {
      const r = await this.chart.build(user, body.prompt, { ip: ctx.ip, userAgent: ctx.userAgent }, body.sessionContext?.lastFilter);
      const meta = { model: r._meta.model, latencyMs: r._meta.latencyMs, parser: r._meta.parser, classifier: 'heuristic' as const, requestId: ctx.requestId };
      return { kind: 'chart', payload: { spec: r.spec, data: r.data, total: r.total }, mode: 'detected' as const, meta };
    }
  }
}
```

```typescript
  constructor(private schema: ZodSchema<T>) {}
```
